mu2p2 = pow(mu2pt*cosh(mu2eta),2),

Float_t mm2e = sqrt(Bp2+Bc_MASS*Bc_MASS)-sqrt(mu1p2+mu1mass*mu1mass)-sqrt(mu2p2+mu2mass*mu2mass)-sqrt(kp2+kmass*kmass),
mm2px = BPt*cos(Bphi)-mu1pt*cos(mu1phi)-mu2pt*cos(mu2phi)-kpt*cos(kphi),

return (Float_t)(mm2e*mm2e-mm2px*mm2px-mm2py*mm2py-mm2pz*mm2pz);
}

auto squaredtransferredmomentum
Initial value:
= [](Float_t Bpt, Float_t Beta, Float_t Bphi, Float_t Bmass,

{

TLorentzVector pmu_B, pmu_mu1, pmu_mu2;

pmu_mu1.SetPtEtaPhiM(mu1pt, mu1eta, mu1phi, mu1mass);

return (Float_t)(pmu_B-pmu_mu1-pmu_mu2).Mag2();
}

= [](Float_t Bpt, Float_t Beta, Float_t Bphi, Float_t Bmass,
Float_t mu1pt, Float_t mu1eta, Float_t mu1phi, Float_t mu1mass,

Float_t Jpsivx, Float_t Jpsivy, Float_t Jpsivz)

Float_t BPt = Bpt*(Bc_MASS/Bmass);
TLorentzVector pmu_vertex, pmu_mu1, pmu_mu2;
pmu_vertex.SetPtEtaPhiM(BPt, vertex.Eta(), vertex.Phi(), Bc_MASS);
pmu_mu1.SetPtEtaPhiM(mu1pt, mu1eta, mu1phi, mu1mass);
pmu_mu2.SetPtEtaPhiM(mu2pt, mu2eta, mu2phi, mu2mass);
return (Float_t)(pmu_vertex-pmu_mu1-pmu_mu2).Mag2();
}
